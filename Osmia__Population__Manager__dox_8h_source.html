1945 virtual void DoLast() {
1946 int today = m_TheLandscape->SupplyDayInYear();
1947 if (today > September) {
1948 int day = g_date->OldDays() + g_date->DayInYear();
1949 double t0 = m_TheLandscape->SupplyTempPeriod(day, 1);
1950
1951 // Check for end of pre-wintering phase
1952 if (!m_PreWinteringEndFlag) {
1953 double t1 = m_TheLandscape->SupplyTempPeriod(day - 1, 1);
1954 double t2 = m_TheLandscape->SupplyTempPeriod(day - 2, 1);
1955 double t3 = m_TheLandscape->SupplyTempPeriod(day - 3, 1);
1956 double t4 = m_TheLandscape->SupplyTempPeriod(day - 4, 1);
1957 double t5 = m_TheLandscape->SupplyTempPeriod(day - 5, 1);
1958
1959 // Sustained autumn cooling pattern
1960 if (((t2 < 13.0) && (t1 < 13.0) && (t0 < 13.0)) &&
1961 (((t5 - t4 > 1.0) && (t4 - t3 > 1.0)) ||
1962 ((t3 < 13.0) && (t5 - t4 >= 3.0)))) {
1963 m_PreWinteringEndFlag = true;
1964 }
1965 }
1966 }
1967
1968 // March 1st marks minimum emergence date
1969 if (today == March) {
1970 m_OverWinterEndFlag = true;
1971 }
1972
1973 // Reset flags after emergence season
1974 if (today == June) {
1975 m_PreWinteringEndFlag = false;
1976 m_OverWinterEndFlag = false;
1977 }
1978
1979#ifdef __OSMIARECORDFORAGE
1980 // Optional foraging statistics output
1981 double meanforage = 0.0;
1982 if (Osmia_Female::m_foragecount > 0)
1983 meanforage = Osmia_Female::m_foragesum / Osmia_Female::m_foragecount;
