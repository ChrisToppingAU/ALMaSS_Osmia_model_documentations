1699 virtual void DoLast() {
1700 int today = m_TheLandscape->SupplyDayInYear();
1701 if (today > September) {
1702 int day = g_date->OldDays() + g_date->DayInYear();
1703 double t0 = m_TheLandscape->SupplyTempPeriod(day, 1);
1704
1705 // Check for end of pre-wintering phase
1706 if (!m_PreWinteringEndFlag) {
1707 double t1 = m_TheLandscape->SupplyTempPeriod(day - 1, 1);
1708 double t2 = m_TheLandscape->SupplyTempPeriod(day - 2, 1);
1709 double t3 = m_TheLandscape->SupplyTempPeriod(day - 3, 1);
1710 double t4 = m_TheLandscape->SupplyTempPeriod(day - 4, 1);
1711 double t5 = m_TheLandscape->SupplyTempPeriod(day - 5, 1);
1712
1713 // Sustained autumn cooling pattern
1714 if (((t2 < 13.0) && (t1 < 13.0) && (t0 < 13.0)) &&
1715 (((t5 - t4 > 1.0) && (t4 - t3 > 1.0)) ||
1716 ((t3 < 13.0) && (t5 - t4 >= 3.0)))) {
1717 m_PreWinteringEndFlag = true;
1718 }
1719 }
1720 }
1721
1722 // March 1st marks minimum emergence date
1723 if (today == March) {
1724 m_OverWinterEndFlag = true;
1725 }
1726
1727 // Reset flags after emergence season
1728 if (today == June) {
1729 m_PreWinteringEndFlag = false;
1730 m_OverWinterEndFlag = false;
1731 }
1732
1733#ifdef __OSMIARECORDFORAGE
1734 // Optional foraging statistics output
1735 double meanforage = 0.0;
1736 if (Osmia_Female::m_foragecount > 0)
1737 meanforage = Osmia_Female::m_foragesum / Osmia_Female::m_foragecount;

1739 Osmia_Female::m_foragesum = 0.0;
1740 Osmia_Female::m_foragecount = 0.0;

1745 if (today == 364) {
1746 ofstream file1("OsmiaStageLengths.txt", ios::app);
