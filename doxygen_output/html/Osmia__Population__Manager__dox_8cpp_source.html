1546
1547 // Set seasonal flags for mid-lifecycle start
1548 m_PreWinteringEndFlag = true;
1549 m_OverWinterEndFlag = false;
1550
1551 // Identify suitable nesting habitat
1552 std::vector<int> suitable_polygons;
1553 m_OurOsmiaNestManager.UpdateOsmiaNesting();
1554 int num_poly = m_TheLandscape->SupplyNumberOfPolygons();
1555 for (int i = 0; i < num_poly; i++) {
1556 if (IsOsmiaNestPossible(i)) {
1557 suitable_polygons.push_back(i);
1558 }
1559 }
1560
1561 int num_poly_for_nesting = suitable_polygons.size();
1562
1563 // Create initial population in parallel
1564 int temp_thread_num = omp_get_max_threads();
1565 int start_num_in_thread = (cfg_OsmiaStartNo.value() / temp_thread_num + 1);
1566
1567 #pragma omp parallel
1568 {
1569 for (int i = 0; i < start_num_in_thread; i++) {
1570 struct_Osmia* sp = new struct_Osmia;
1571 sp->OPM = this;
1572 sp->L = m_TheLandscape;
1573
1574 // Assign random mass within configured range
1575 double minmass = (cfg_OsmiaFemaleMassMin.value() - 4) / 0.25;
1576 double maxmass = (cfg_OsmiaFemaleMassMax.value() - 4) / 0.25;
1577 sp->mass = minmass + (maxmass - minmass) * g_rand_uni_fnc();
1578
1579 sp->parasitised = TTypeOfOsmiaParasitoids::topara_Unparasitised;
1580 sp->sex = true; // All females (males not modelled)
1581
1582 // Random placement in suitable habitat
1583 int pindex = suitable_polygons[g_random_fnc(num_poly_for_nesting)];
1584 APoint temp_point = m_TheLandscape->SupplyARandomLocPoly(pindex);
1585 sp->x = temp_point.m_x;
1586 sp->y = temp_point.m_y;
1587 sp->nest = CreateNest(sp->x, sp->y, pindex);
1588
1589 // Set initial overwintering progress
1590 sp->overwintering_degree_days = cfg_OsmiaOverwinterDegreeDaysInitialSimu.value();
1591
1592 // Create InCocoon individual
1593 CreateObjects(TTypeOfOsmiaLifeStages::to_OsmiaInCocoon, NULL, sp, 1);
1594 delete sp;
1595 }
1596 }
1597
1598 // Set age for all created InCocoon individuals
1599 for (unsigned co = 0; co < unsigned(SupplyListSize(int(TTypeOfOsmiaLifeStages::to_OsmiaInCocoon))); co++) {
1600 dynamic_cast<Osmia_InCocoon*>(SupplyAnimalPtr(int(TTypeOfOsmiaLifeStages::to_OsmiaInCocoon), co))->SetAgeDegrees(2000);
1601 }
1602
1603 // Cache frequently-accessed parameters
1604 m_PollenCompetitionsReductionScaler = cfg_OsmiaDensityDependentPollenRemovalConst.value();
1605
1606 // Populate prepupal development rate lookup table
1607 m_PrePupalDevelRates.resize(42);
1608 for (int i = 0; i < 42; i++) {
1609 m_PrePupalDevelRates[i] = cfg_OsmiaPrepupalDevelRates.value(i);
1610 }
1611

1613 m_is_paralleled = true;
1614

1616#ifdef __OSMIA_PESTICIDE_STORE
1617 ofstream oversprayfile("osmia_overspray.txt", ios::trunc);
1618 oversprayfile << "Year" << '\t' << "Day" << '\t' << "Female ID"
