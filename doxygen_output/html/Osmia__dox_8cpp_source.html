1639void Osmia_Egg::Step(void)
1640{
1644 if (m_StepDone || m_CurrentStateNo == -1) return;
1645 switch (m_CurrentOState)
1646 {
1647 case toOsmias_InitialState: // Initial state always starts with develop
1648 m_CurrentOState = toOsmias_Develop;
1649 break;
1650 case toOsmias_Develop:
1651 m_CurrentOState = st_Develop();
1652 m_StepDone = true;
1653 break;
1654 case toOsmias_NextStage:
1655 m_CurrentOState = st_Hatch();
1656 break;
1657 case toOsmias_Die:
1658 st_Dying(); // No return value - no behaviour after this
1659 m_StepDone = true;
1660 break;
1661 default:
1662 m_OurLandscape->Warn("Osmia_Egg::Step()", "unknown state - default");
1663 std::exit(TOP_Osmia);
1664 }
1665}

1709TTypeOfOsmiaState Osmia_Egg::st_Develop(void)
1710{
1711 /*
1712 * Development is preceded by a mortality test, then a day degree calculation is made to determine the development that occured in the last 24 hours.
1713 * When enough day degrees are achieved the egg hatches.If it does not hatch then the development behaviour is queued up for the next day.
1714 */
1715 if (!m_OurNest->GetIsOpen()){
1716 if (DailyMortality()) return toOsmias_Die;
1717 //killed by pesticide
1718 #ifdef __OSMIA_PESTICIDE_ENGINE
1719 if(cfg_OsmiaEggThresholdBasedPesticideResponse.value()){
1720 if (g_rand_uni_fnc()<m_egg_pest_mortality){
1721 return toOsmias_Die;
1722 }
1723 else{
1724 m_egg_pest_mortality = 0; //only die ones, otherwise set it to 0
1725 }
1726 }
1727
1728 #endif
1729 }
1730 m_Age++;
1731 double DD = m_TempToday- m_OsmiaEggDevelThreshold;
1732 if (DD > 0) m_AgeDegrees += DD;
1733 if (m_AgeDegrees > m_OsmiaEggDevelTotalDD) return toOsmias_NextStage;
1734 return toOsmias_Develop;
1735}

1936TTypeOfOsmiaState Osmia_Larva::st_Develop(void)
1937{
1938 if (!m_OurNest->GetIsOpen())
1939 if (DailyMortality()) return toOsmias_Die;
1940 m_Age++;

1944 return toOsmias_Develop;

2706TTypeOfOsmiaState Osmia_InCocoon::st_Develop(void)
2707{
2714 m_Age++;

2716 {
2717 // Must be after pre-wintering
2718 if (!m_OurPopulationManager->IsOverWinterEnd())
2719 {

2721 double DD = m_TempToday - m_OsmiaInCocoonOverwinteringTempThreshold;
2722 if (DD > 0) m_AgeDegrees += DD;
2723 }
2724 else // It is >= March 1st
2725 {
2726 if (m_DayInYear == March+1) { // if first day of March
2727 m_emergencecounter = int(m_OsmiaInCocoonEmergCountConst + m_OsmiaInCocoonEmergCountSlope * m_AgeDegrees) + m_emergenceday.Geti() + m_OurNest->GetAspectDelay();
2728 }

2730 {
2731 if (--m_emergencecounter < 1)
2732 {
2733 if (WinterMortality()) return toOsmias_Die; // a once only test for overwintering mortality
2734 else return toOsmias_NextStage;
2735 }
2736
2737 if(m_DayInYear == June-1){//too late to emerge
2738 return toOsmias_Die;
2739 }
2740 }
2741 }
2742 }
2743 else
2744 {
2745 // Must be pre-wintering so count up prewintering day degrees
2746 if (m_TempToday > m_OsmiaInCocoonPrewinteringTempThreshold) m_DDPrewinter += (m_TempToday - m_OsmiaInCocoonPrewinteringTempThreshold);
